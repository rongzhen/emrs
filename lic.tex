\begin{frame}
  \frametitle{Current results:  quantum efficiency in Cu-based materials solar cells}
  Optical efficiency in each layer:
  \begin{equation}
     A(d_i)=\frac{\int I(\lambda) QE(\lambda, d_i) d\lambda}{\int I(\lambda) d\lambda}
  \end{equation}

    \begin{equation}
     A'(d_i)=\frac{\int I(\lambda) QE(\lambda, d_i) d\lambda}{\int I(\lambda) QE(\lambda, d_i = \infty) d\lambda}
  \end{equation}
  
  Here, I($\lambda$) is the spectral power density of the AM1.5 solar spectrum with wavelength $\lambda$. QE($\lambda, d_i$) is the quantum efficiency in the corresponding layer with thickness $d_i$.
\end{frame}
\end{comment}

\begin{frame}
\frametitle{Summary for current work }
\begin{itemize}
 
 \item Cu\textsubscript{x}(Bi,Sb)(Se,S)\textsubscript{y} and SnS materials have high absorption coefficient compared with CuZnSn(Se,S)\textsubscript{4}, 
 and Cu\textsubscript{m}SnS\textsubscript{n} materials have similar absorption coefficient compared with CuZnSn(Se,S)\textsubscript{4}.

 \item The high absorption coefficients can be explained by the flat valence and conduction bands, as well as stronger $p$-states of Bi, Sb for Cu\textsubscript{x}(Bi,Sb)(Se,S)\textsubscript{y} 
 and Sn for SnS.
 
 \item GW and HSE are valuable methods to calculate band-gap for semiconductor. The absolute value of band-gap error is around 0.2 eV compared with experimental values. 




\end{itemize}



\end{frame}
